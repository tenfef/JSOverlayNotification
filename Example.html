Example for JSOverlayNotification

$(function(){
	overlay_notification.show_overlay_notification({
		text : 'The world is going to end.',
		unique_id : "notification_id",
		buttons : [
			{
				title : "Click Me",
				callback : function(){
					alert('First button click');
				}
			},
			{
				title : "Second Button",
				callback : function(){
					alert('Second Button Click');
				}
			}
		]
	});
});
